     element: HTMLElement;
     scene: THREE.Scene;
     camera: THREE.Camera;
     renderFn: (scene: THREE.Scene, camera: THREE.Camera) => void;
     visible: boolean;
   };

3. **Exported functions:**
   - `getState(): SignalCanvasState` — accessor for the globalThis singleton
   - `initSignalCanvas(canvas: HTMLCanvasElement): void` — creates WebGLRenderer if not exists, sets up GSAP ticker + reduced-motion listener + resize handler

   - `renderer.setScissorTest(false)` then `renderer.clear()` then `renderer.setScissorTest(true)`
   - For each visible scene entry: get `getBoundingClientRect()`, compute Y-inverted coords: `const canvasY = renderer.domElement.clientHeight - rect.bottom`

```typescript
"use client";

/**
 * Hook for registering a Three.js scene with the SignalCanvas singleton.
 * Handles: scene registration, IntersectionObserver visibility gating,
 * GPU resource disposal on unmount.
 *
 * Usage:
 *   const containerRef = useRef<HTMLDivElement>(null);
 *   useSignalScene(containerRef, () => ({
 *     scene: new THREE.Scene(),
 *     camera: new THREE.PerspectiveCamera(75, 1, 0.1, 100),
 *   }));
 *
 * @module hooks/use-signal-scene
 */

import { useEffect, useRef } from "react";
import type * as THREE from "three";
import {
  registerScene,
  deregisterScene,
  setSceneVisibility,
  disposeScene,
} from "@/lib/signal-canvas";

type SceneBuildResult = {
  scene: THREE.Scene;
  camera: THREE.Camera;
  /** Optional custom render function. Default: renderer.render(scene, camera) */
  renderFn?: (scene: THREE.Scene, camera: THREE.Camera) => void;
};

/**
 * Register a Three.js scene with the SignalCanvas singleton.
 *
 * @param elementRef - Ref to the DOM element that defines the scene's viewport rectangle
 * @param buildScene - Factory function called once on mount to create scene + camera
 */
export function useSignalScene(
  elementRef: React.RefObject<HTMLElement | null>,
  buildScene: () => SceneBuildResult
): void {
  const idRef = useRef(crypto.randomUUID());

  useEffect(() => {
    const element = elementRef.current;
    if (!element) return;

    const { scene, camera, renderFn } = buildScene();
    const id = idRef.current;

    // IntersectionObserver gates rendering — offscreen scenes skip render loop
    const observer = new IntersectionObserver(
      ([entry]) => {
        setSceneVisibility(id, entry.isIntersecting);
      },
      { threshold: 0 }
    );
    observer.observe(element);

    // Register with singleton — rendering is handled by the singleton's GSAP ticker
    registerScene(id, {
      element,
      scene,
      camera,
      renderFn,
      visible: true,
    });

    return () => {
      observer.disconnect();
      deregisterScene(id);
      disposeScene(scene); // GPU cleanup: geometry, material, texture disposal
    };
  }, []); // eslint-disable-line react-hooks/exhaustive-deps
  // intentionally empty deps — runs once per mount, buildScene is stable
}
```

Key design decisions:
- `crypto.randomUUID()` for scene ID — unique per hook instance, stable across renders
- `IntersectionObserver` with `threshold: 0` — scene is visible as soon as any pixel enters viewport
- Cleanup calls `deregisterScene` (removes from singleton Map) AND `disposeScene` (GPU resource traversal)
- `buildScene` is called inside useEffect (not at render time) to avoid creating Three.js objects during SSR or before mount
- `type` import for `THREE` namespace — avoids pulling three into the module's own chunk; actual THREE runtime comes from signal-canvas.ts
- The `renderFn` parameter is optional — default in registerScene falls back to `renderer.render(scene, camera)`
