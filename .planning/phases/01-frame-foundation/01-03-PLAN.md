```css
/* ══════════════════════════════════════════════════════════════
   PRINT STYLES
   Invert dark backgrounds · suppress Signal layer · preserve layout
   ══════════════════════════════════════════════════════════════ */

@media print {
  /* ── Suppress Signal layer completely ── */
  .vhs-overlay,
  .vhs-crt,
  .vhs-noise,
  .vhs-scanlines,
  [data-anim],
  .sf-cursor,
  .sf-grain::after,
  .sf-idle-overlay,
  .gsap-marker-start,
  .gsap-marker-end,
  .gsap-marker-scroller-start,
  .gsap-marker-scroller-end {
    display: none !important;
  }

  /* ── Invert dark surfaces for paper ── */
  body,
  .dark body,
  .dark {
    background: white !important;
    color: black !important;
  }

  /* ── Force readable contrast on all text ── */
  * {
    color: black !important;
    -webkit-print-color-adjust: exact;
    print-color-adjust: exact;
  }

  /* ── Preserve layout structure ── */
  .sf-border {
    border-color: black !important;
  }

  /* ── Hide interactive-only elements ── */
  button,
  [role="button"],
  nav,
  .scroll-to-top {
    display: none !important;
  }

  /* ── Links show URL for reference ── */
  a[href]::after {
    content: " (" attr(href) ")";
    font-size: 0.8em;
    color: #666 !important;
  }

  /* ── Remove fixed positioning ── */
  * {
    position: static !important;
  }

  /* ── Ensure content fills page width ── */
  body {
    max-width: 100% !important;
    margin: 0 !important;
    padding: 1cm !important;
  }
}
```
