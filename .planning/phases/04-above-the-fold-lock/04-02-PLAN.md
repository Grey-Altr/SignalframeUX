**Part A: Rewrite app/error.tsx as FRAME+SIGNAL moment**

Rewrite error.tsx keeping `"use client"` directive. Structure:

```tsx
"use client";

import { useEffect } from "react";
import { SFContainer } from "@/components/sf";
import { SFText } from "@/components/sf";
import { SFButton } from "@/components/sf";

export default function Error({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  useEffect(() => {
    // Guard: skip ScrambleText on reduced-motion
    if (window.matchMedia("(prefers-reduced-motion: reduce)").matches) return;

    const el = document.querySelector("[data-anim='error-code']");
    if (!el) return;

    import("@/lib/gsap-plugins").then(({ gsap }) => {
      gsap.to(el, {
        duration: 0.6,
        scrambleText: {
          text: "ERROR",
          chars: "01!<>-_\\/[]{}—=+*^?#",
          speed: 0.5,
        },
      });
    });
  }, []);

  return (
    <main id="main-content" className="min-h-screen flex items-center justify-center bg-background px-6">
      <SFContainer>
        <div role="alert" className="text-center">
          <h1 className="sr-only">Error</h1>
          <div
            data-anim="error-code"
            aria-hidden="true"
            className="sf-display sf-glitch text-[clamp(48px,8vw,80px)] text-destructive mb-4"
          >
            ERROR
          </div>
          <SFText as="p" intent="body" className="uppercase tracking-[0.1em] text-muted-foreground mb-6">
            AN UNEXPECTED ERROR OCCURRED
          </SFText>
          {error.digest && (
            <SFText as="p" intent="small" className="font-mono uppercase tracking-wider text-muted-foreground mb-8">
              DIGEST: {error.digest}
            </SFText>
          )}
          <SFButton intent="signal" onClick={reset}>
            TRY AGAIN
          </SFButton>
        </div>
      </SFContainer>
    </main>
  );
}
```

Key decisions:
- SFContainer for FRAME structure (enforces max-width + gutters)
- SFText for body/small text (enforces semantic typography)
- `sf-glitch` class on error code for VHS glitch effect (already has reduced-motion guard in CSS)
- `data-anim="error-code"` for GSAP ScrambleText targeting
- ScrambleText guarded with `prefers-reduced-motion` check BEFORE async import
- `sf-display` class for the large error code (existing display font class)
- Spacing uses blessed stops: mb-4 (16px), mb-6 (24px), mb-8 (32px)
- Error tone: DU/TDR — terminal-like, not apologetic
