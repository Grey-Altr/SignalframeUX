    **Exact transformation — AFTER:**

    ```typescript
    "use client";

    import { forwardRef, useRef, useImperativeHandle, useEffect, type ReactNode } from "react";
    import { gsap, ScrollTrigger } from "@/lib/gsap-core";

    interface PinnedSectionProps {
      children: ReactNode;
      className?: string;
      /** Total scroll distance in viewport heights (e.g., 2 = 200vh, 3 = 300vh) */
      scrollDistance: number;
      id?: string;
    }

    /**
     * PinnedSection — reusable pin/scrub scroll wrapper.
     *
     * Pins this section to the viewport while the user scrolls through
     * `scrollDistance` viewport heights of content. Consumed by Phase 31
     * (THESIS manifesto) and Phase 32 (SIGNAL section).
     *
     * Exposes its root container via `React.forwardRef` so consumers can
     * use it as the `pinnedContainer` for nested ScrollTriggers that need
     * to scrub their own timelines inside the pinned span.
     *
     * - Reduced-motion: section renders statically, no ScrollTrigger created.
     * - Cleanup: gsap.context().revert() destroys all created ScrollTriggers.
     * - Root div has no overflow:hidden — GSAP pin switches to position:fixed,
     *   clipping parents break the pin geometry.
     *
     * @example
     * ```tsx
     * const pinnedRef = useRef<HTMLDivElement>(null);
     * <PinnedSection ref={pinnedRef} scrollDistance={2.5} id="thesis">
     *   <ManifestoContent />
     * </PinnedSection>
     * ```
     */
    export const PinnedSection = forwardRef<HTMLDivElement, PinnedSectionProps>(
      function PinnedSection(
        { children, className, scrollDistance, id },
        forwardedRef
      ) {
        const containerRef = useRef<HTMLDivElement>(null);

        // Merge the forwarded ref with our internal ref so ScrollTrigger has a stable
        // target AND consumers can also read the DOM node via the forwarded ref.
        useImperativeHandle(
          forwardedRef,
          () => containerRef.current as HTMLDivElement,
          []
        );

        useEffect(() => {
          const container = containerRef.current;
          if (!container) return;

          // prefers-reduced-motion: instant placement, no scroll-driven animation
          if (window.matchMedia("(prefers-reduced-motion: reduce)").matches) return;

          const ctx = gsap.context(() => {
            ScrollTrigger.create({
              trigger: container,
              pin: true,
              scrub: 1,
              anticipatePin: 1,
              start: "top top",
              end: () => `+=${scrollDistance * window.innerHeight}`,
              invalidateOnRefresh: true,
            });
          });

          return () => ctx.revert();
        }, [scrollDistance]);

        return (
          <div
            ref={containerRef}
            id={id}
            className={className}
            style={{ height: "100vh" }}
          >
            {children}
          </div>
        );
      }
    );
    ```

    **Why `useImperativeHandle` instead of passing the forwarded ref directly to the div:** The internal `ScrollTrigger.create` call depends on `containerRef.current` being stable across renders. If we passed `forwardedRef` directly to the div, we would need runtime null-checks for both a function ref and an object ref. `useImperativeHandle` cleanly bridges them: the internal ref is the source of truth for the effect, and the forwarded ref receives a synchronous mirror via `useImperativeHandle` (React calls it after commit, before the effect runs — this is safe for the consumer's `useGSAP` hook because `useGSAP` also runs after commit).

    ```typescript
    "use client";

    import { useRef } from "react";
    import type { CSSProperties } from "react";
    import { gsap, SplitText, useGSAP } from "@/lib/gsap-split";
    import type { ManifestoStatementData } from "@/lib/thesis-manifesto";

    interface ManifestoStatementProps extends ManifestoStatementData {
      /** Parent master timeline — each statement adds its own enter/hold/exit tweens */
      timeline: gsap.core.Timeline;
      /** Timeline position label or numeric offset where this statement's enter tween lands */
      position: number | string;
      /** Void (vh) before this statement's position — exposed as data-void-before for TH-04 test */
      voidBefore: number;
    }

    const STATEMENT_STYLE: CSSProperties = {
      fontSize: "clamp(56px, 10vw, 120px)",
      lineHeight: 0.9,
      letterSpacing: "-0.02em",
    };

    const STATEMENT_CLASS = "font-[family-name:var(--font-display)] text-foreground";

    /**
     * ManifestoStatement — single THESIS manifesto anchor statement.
     *
     * Renders an absolutely-positioned container at its anchor coordinates,
     * with a child span whose characters are split and choreographed via
     * the parent master timeline (scrubbed by the THESIS ScrollTrigger).
     *
     * All statements are display-scale anchors (Anton clamp(56px, 10vw, 120px)).
     * No connector tier exists — the wiki-locked 6-statement set is all-anchor.
     *
     * - Reduced-motion: parent ThesisSection short-circuits before rendering this tree
     * - D-13 compliance: tweens target self.chars (grandchildren of pinned element), never the root
     * - D-26 compliance: no data-anim attribute on any element
     */
    export function ManifestoStatement({
      text,
      anchor,
      mobileAnchor,
      timeline,
      position,
      voidBefore,
    }: ManifestoStatementProps) {
      const rootRef = useRef<HTMLDivElement>(null);
      const spanRef = useRef<HTMLSpanElement>(null);

      useGSAP(
        () => {
          if (!spanRef.current) return;
          // Defense in depth — parent also guards, but skip init if reduced-motion is active
          if (window.matchMedia("(prefers-reduced-motion: reduce)").matches) return;

          SplitText.create(spanRef.current, {
            type: "chars",
            mask: "chars",
            autoSplit: true,
            onSplit(self: { chars: Element[] }) {
              timeline
                .from(
                  self.chars,
                  {
                    yPercent: 100,
                    opacity: 0,
                    duration: 0.5,
                    stagger: 0.02,
                    ease: "sf-snap",
                  },
                  position,
                )
                .to({}, { duration: 0.3 }, ">") // hold window
                .to(
                  self.chars,
                  { opacity: 0, duration: 0.3, ease: "power1.in" },
                  ">",
                );
            },
          });
        },
        { scope: rootRef, dependencies: [text, position] },
      );

      // Resolve anchor — use mobileAnchor on <=667px viewports if provided
      const isMobile =
        typeof window !== "undefined" &&
        window.matchMedia("(max-width: 667px)").matches;
      const resolvedAnchor = isMobile && mobileAnchor ? mobileAnchor : anchor;

      return (
        <div
          ref={rootRef}
          className="absolute rounded-none"
          style={resolvedAnchor as CSSProperties}
          data-void-before={voidBefore}
        >
          <span
            ref={spanRef}
            data-statement
            data-statement-size="anchor"
            className={STATEMENT_CLASS}
            style={STATEMENT_STYLE}
          >
            {text}
          </span>
        </div>
      );
    }
    ```

    ```typescript
    "use client";

    import { useRef, useState, useEffect } from "react";
    import { gsap, ScrollTrigger, useGSAP } from "@/lib/gsap-split";
    import { PinnedSection } from "@/components/animation/pinned-section";
    import { ManifestoStatement } from "@/components/blocks/manifesto-statement";
    import { THESIS_MANIFESTO } from "@/lib/thesis-manifesto";

    // Per-statement scroll budget in the master timeline — each statement gets 1 timeline "unit"
    // made up of enter (0.5) + hold (0.3) + exit (0.3) = 1.1 seconds scaled by scrub.
    // We use integer timeline positions (0, 1, 2, ...) so the scrubbed master lines up 1:1 with
    // THESIS_MANIFESTO order. Actual duration is normalized by ScrollTrigger scrub.
    const TIMELINE_UNIT = 1.1;

    // Weighted-arc void budget (vh) exposed to DOM for TH-04 assertion.
    // S1 (signal-frame opener) and S6 (signal-frame closer) are the structural bookends and get
    // extra breathing room; S2-S5 share a tighter rhythm. Index-based lookup by THESIS_MANIFESTO position.
    const BOOKEND_VOID_VH = 40;   // indices 0 and 5 → S1, S6
    const INTERIOR_VOID_VH = 25;  // indices 1-4 → S2, S3, S4, S5

    const voidBeforeFor = (index: number, total: number): number =>
      index === 0 || index === total - 1 ? BOOKEND_VOID_VH : INTERIOR_VOID_VH;

    /**
     * ThesisSection — 200-300vh pinned scroll manifesto.
     *
     * Composes Phase 29's PinnedSection primitive with a nested scrubbed
     * ScrollTrigger that scrubs a single master timeline containing per-statement
     * enter/hold/exit windows. Each statement is rendered by ManifestoStatement
     * with its own SplitText init.
     *
     * Decisions:
     * - D-13: Tweens target statement chars only — never the PinnedSection root or stage div
     * - D-19: scrollDistance = 2.5 desktop, 2 mobile (D-31)
     * - D-21: Nested ScrollTrigger with pinnedContainer (not extending PinnedSection API)
     * - D-22: Block component, 'use client', no data-anim attribute (D-26)
     * - D-27-D-30: Reduced-motion branch returns stacked specimen with all statements in flow
     *
     * Consumed by app/page.tsx inside the THESIS SFSection landmark.
     */
    export function ThesisSection() {
      const pinnedRef = useRef<HTMLDivElement>(null);
      const stageRef = useRef<HTMLDivElement>(null);

      // Detect mobile once at mount — do not listen for resize; ScrollTrigger.refresh handles layout changes
      const [scrollDistance, setScrollDistance] = useState<number>(2.5);
      const [reducedMotion, setReducedMotion] = useState<boolean>(false);

      useEffect(() => {
        const mobileMQ = window.matchMedia("(max-width: 667px)");
        const rmMQ = window.matchMedia("(prefers-reduced-motion: reduce)");
        setScrollDistance(mobileMQ.matches ? 2 : 2.5);
        setReducedMotion(rmMQ.matches);
      }, []);

      // Compute per-statement timeline positions and weighted-arc void-before values
      const statementEntries = THESIS_MANIFESTO.map((s, i) => {
        const position = i * TIMELINE_UNIT;
        const voidBefore = voidBeforeFor(i, THESIS_MANIFESTO.length);
        return { statement: s, position, voidBefore, key: `${i}-${s.pillar}` };
      });

      useGSAP(
        () => {
          // Motion branch only — reduced-motion branch returns early before this runs
          if (reducedMotion) return;
          if (!stageRef.current || !pinnedRef.current) return;

          const tl = gsap.timeline({ paused: true });

          // ManifestoStatement children add their own tweens to this timeline via props.
          // The nested ScrollTrigger below scrubs the timeline against the parent pin window.

          ScrollTrigger.create({
            trigger: stageRef.current,
            pinnedContainer: pinnedRef.current,
            start: "top top",
            end: "bottom bottom",
            scrub: 1,
            animation: tl,
            invalidateOnRefresh: true,
          });
        },
        { scope: stageRef, dependencies: [reducedMotion, scrollDistance] }
      );

      // Reduced-motion fallback — stacked specimen, no pin, no scrub, no SplitText.
      // All 6 statements render at display scale in document order.
      if (reducedMotion) {
        return (
          <div
            data-thesis-reduced-motion
            className="flex flex-col gap-24 py-24 px-6 md:px-16"
          >
            {THESIS_MANIFESTO.map((s, i) => (
              <div
                key={`${i}-${s.pillar}`}
                className="text-left rounded-none"
                data-void-before={voidBeforeFor(i, THESIS_MANIFESTO.length)}
              >
                <span
                  data-statement
                  data-statement-size="anchor"
                  className="font-[family-name:var(--font-display)] text-foreground"
                  style={{
                    fontSize: "clamp(56px, 10vw, 120px)",
                    lineHeight: 0.9,
                    letterSpacing: "-0.02em",
                  }}
                >
                  {s.text}
                </span>
              </div>
            ))}
          </div>
        );
      }

      // Motion branch — pinned scroll choreography
      return (
        <PinnedSection
          ref={pinnedRef}
          scrollDistance={scrollDistance}
          id="thesis-pin"
        >
          <div
            ref={stageRef}
            data-pinned-container
            data-stage
            className="relative w-full h-screen overflow-hidden"
          >

                {...statement}

                position={position}
                voidBefore={voidBefore}
              />
            ))}
          </div>
        </PinnedSection>
      );
    }

The real pattern is this: create the timeline inside `useGSAP`, then use a `useRef<gsap.core.Timeline>(null)` at the ThesisSection level to hold the timeline instance, and pass `tlRef` (not `tl`) to children. Children read `tlRef.current` inside their own `useGSAP`. This works because all `useGSAP` hooks run after commit and after the parent's `useGSAP` has populated the ref.

    **Revised pattern — use this instead of the sketch above:**

    1. At ThesisSection top: `const tlRef = useRef<gsap.core.Timeline | null>(null);`
    2. Inside parent `useGSAP`: `tlRef.current = gsap.timeline({ paused: true });` followed by the `ScrollTrigger.create` with `animation: tlRef.current`.
    3. Pass `tlRef` to `ManifestoStatement` via a new prop `timelineRef: React.RefObject<gsap.core.Timeline | null>`.
    4. Inside `ManifestoStatement`'s `useGSAP`, read `const tl = timelineRef.current; if (!tl) return;` before calling `SplitText.create`.

    This ordering works because:
    - Parent `useGSAP` runs first (its scope is `stageRef` which is the parent DOM node).
