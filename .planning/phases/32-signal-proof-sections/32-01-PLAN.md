<objective>
Deliver the PROOF homepage section — a full-viewport interactive demonstration of SIGNAL/FRAME layer separation driven by pointer position (desktop), touch drag (mobile), and device tilt (iOS/Android gyroscope). Three concurrent layers (A: shader branch, B: component skeletons, C: FRAME-pole column) read a single section-scoped `--signal-intensity` CSS custom property driven by a rAF-smoothed lerp loop.

**PROOF is the higher-risk half of Phase 32.** Building it first forces the hard architectural decisions (section-scoped custom property bridging, rAF lerp loop, SignalCanvas 3rd-scene registration, iOS permission gate) before SIGNAL's lower-risk atmospheric work in Plan 02. The locked cdBrain decision is A+B+C simultaneously — not alternatives, not before/after states. At `intensity=0` all three layers are fully visible (lattice shader + bright skeletons + FRAME-pole data). At `intensity=1` the shader reaches full generative output, skeletons fade to zero opacity, and the FRAME-pole column remains as structural reference (its opacity is NOT driven — only the skeletons fade). The result reads as two live states of one system, not a reveal panel.

Purpose: Phase 32 closes the interactive pillar of the v1.5 homepage. PROOF is the "does a non-designer feel something when they move their mouse?" test — the quality bar is emotional, not cognitive. The user must discover the interaction in <=5 seconds without any instruction overlay, on every input surface.

- `lib/proof-components.ts` — the 12-item PROOF_COMPONENT_SKELETONS constant (shared with Phase 33 INVENTORY)
- `components/animation/proof-shader.tsx` — WebGL shader fork of GLSLHero with geometric-lattice branch and section-scoped MutationObserver
- `components/animation/component-skeleton.tsx` — stroke-only SF component skeletons (Approach B)
- `components/blocks/proof-section.tsx` — the block composing A+B+C with rAF lerp, pointermove, gyroscope permission gate, ScrollTrigger lifecycle
- `app/page.tsx` — imports and renders <ProofSection /> inside the existing #proof SFSection, replacing the stub

          const result = await DOE.requestPermission();
          return result === "granted";
        } catch {
          return false;
        }
      }
      // Android / non-iOS: no permission required, events fire immediately
      return true;
    }

    function clampGamma(gamma: number | null): number {
      if (gamma === null) return 0.5;
      // Clamp to -60..+60 to avoid ±90 discontinuity
      const clamped = Math.max(-60, Math.min(60, gamma));
      return (clamped + 60) / 120;
    }

    export function ProofSection() {
      const sectionRef = useRef<HTMLElement>(null);
      const skeletonRef = useRef<HTMLDivElement>(null);

      useGSAP(
        () => {
          const section = sectionRef.current;
          if (!section) return;

          // Reduced-motion: render static split, no animation, no listeners
          if (window.matchMedia("(prefers-reduced-motion: reduce)").matches) {
            section.style.setProperty("--signal-intensity", "0.5");
            if (skeletonRef.current) {
              skeletonRef.current.style.opacity = "0.5";
            }
            return;
          }

          // Default state — restore to SIGNAL (1.0) on leave
          _targetIntensity = 0.8;
          _currentIntensity = 0.8;
          section.style.setProperty("--signal-intensity", "0.8");

          // ── Pointer handler — desktop + touch Pointer Events ─────────────
          const handlePointerMove = (e: PointerEvent) => {
            _targetIntensity = Math.max(
              0,
              Math.min(1, e.clientX / window.innerWidth)
            );
          };
          const handlePointerLeave = () => {
            _targetIntensity = 1.0;
          };

          // ── Gyroscope handler (attached after iOS permission grant) ─────
          let gyroAttached = false;
          const handleDeviceOrientation = (e: DeviceOrientationEvent) => {
            _targetIntensity = clampGamma(e.gamma);
          };
          const attachGyro = () => {
            if (gyroAttached) return;
            gyroAttached = true;
            window.addEventListener(
              "deviceorientation",
              handleDeviceOrientation,
              { passive: true }
            );
          };

            passive: true,
          });

              passive: true,
            });
            section.addEventListener("pointerleave", handlePointerLeave, {
              passive: true,
            });
          };

            section.removeEventListener("pointermove", handlePointerMove);
            section.removeEventListener("pointerleave", handlePointerLeave);
          };

          const trigger = ScrollTrigger.create({
            trigger: section,
            start: "top bottom",
            end: "bottom top",
            invalidateOnRefresh: true,
            onEnter: () => {
              activatePointerListener();
              startLerpLoop(section, skeletonRef.current);
            },
            onLeave: () => {
              deactivatePointerListener();
              stopLerpLoop();
              section.style.setProperty("--signal-intensity", "1.0");
              if (skeletonRef.current) {
                skeletonRef.current.style.opacity = "0";
              }
            },
            onEnterBack: () => {
              activatePointerListener();
              startLerpLoop(section, skeletonRef.current);
            },
            onLeaveBack: () => {
              deactivatePointerListener();
              stopLerpLoop();
              section.style.setProperty("--signal-intensity", "1.0");
              if (skeletonRef.current) {
                skeletonRef.current.style.opacity = "0";
              }
            },
          });
          // Zero-range trap diagnostic (RESEARCH pitfall 1 — adapted for onEnter triggers)
          console.debug(
            "[PROOF ST]",
            "start:",
            trigger.start,
            "end:",
            trigger.end
          );

          return () => {
            deactivatePointerListener();
            stopLerpLoop();
            if (gyroAttached) {
              window.removeEventListener(
                "deviceorientation",
                handleDeviceOrientation
              );
            }
            section.removeEventListener("touchstart", onFirstTouch);
            trigger.kill();
          };
        },
        { scope: sectionRef, dependencies: [] }
      );

      return (
        <section
          ref={sectionRef as React.RefObject<HTMLElement>}
          id="proof"
          data-section="proof"
          data-anim
          style={{ height: "100vh", position: "relative", overflow: "hidden" }}
          className="rounded-none"
        >

          <ProofShader
            sectionRef={sectionRef as React.RefObject<HTMLElement | null>}
          />

          {/* Layer B — component skeletons (absolute inset-0 z-10) */}
          <SkeletonGrid ref={skeletonRef} />

          {/* Layer C — FRAME-pole left column (absolute left, z-20) */}
          <div
            data-proof-layer="frame-pole"
            className={cn(
              "absolute left-0 top-0 bottom-0 z-20",
              "w-96 p-32 flex flex-col gap-24",
              "font-mono text-sm text-foreground rounded-none",
              "bg-background/0"
            )}
          >
            <div className="flex flex-col gap-8">
              <span className="text-2xs uppercase tracking-widest opacity-60">
                SYSTEM READOUT
              </span>
              <span>COMPONENTS // 51</span>
              <span>BUNDLE // 100KB</span>
              <span>LIGHTHOUSE // 100/100</span>
            </div>
            <div className="flex flex-col gap-4">
              <span className="text-2xs uppercase tracking-widest opacity-60">

              </span>

                <span key={id}>{id}</span>
              ))}

          </div>
        </section>
      );
    }
