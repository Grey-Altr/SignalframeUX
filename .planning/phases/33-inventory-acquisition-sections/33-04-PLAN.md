Create the companion client component for the copy button:

```tsx
// components/blocks/acquisition-copy-button.tsx
'use client'

import { useState, useCallback } from 'react'

interface AcquisitionCopyButtonProps {
  command: string
}

export function AcquisitionCopyButton({ command }: AcquisitionCopyButtonProps) {
  const [copied, setCopied] = useState(false)

  const handleCopy = useCallback(async () => {
    try {
      await navigator.clipboard.writeText(command)
      setCopied(true)
      setTimeout(() => setCopied(false), 1500)
    } catch {
      // Silent fail — clipboard may be unavailable (e.g. non-HTTPS dev)
    }
  }, [command])

  return (
    <span
      data-copy-trigger
      role="button"
      tabIndex={0}
      aria-label={`Copy: ${command}`}
      onClick={handleCopy}
      onKeyDown={e => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); handleCopy() } }}
      className="font-mono text-xs text-muted-foreground hover:text-foreground cursor-pointer transition-colors duration-[34ms] select-none"
    >
      {copied ? '[COPIED]' : '[COPY]'}
    </span>
  )
}
```

**Important constraints:**
- `data-acquisition-root` on the section root (required by AQ-04 test)
- `data-copy-trigger` on the copy button (required by AQ-01 test)
- `style={{ maxHeight: '50vh' }}` on section root (hard cap)
- `overflow-hidden` — if content exceeds 50vh, it clips, does not grow
- Zero `rounded`, zero `border-radius`, zero `SFButton`, zero "Get Started"
- Padding is `py-8 md:py-10` (compact) — the ≤50vh cap requires restraint on vertical padding
- The CLI hero text size: `text-xl md:text-2xl lg:text-3xl` — large enough to dominate, but size the column so it fits within 50vh
