    ```tsx
    "use client";

    import { useRef, type KeyboardEvent } from "react";

    type ColorSwatch = { step: number; l: number; c: number; h: number };
    type ColorScale = { name: string; hue: number; swatches: ColorSwatch[] };

    interface ColorSpecimenProps {
      scales: ColorScale[];
      showAll: boolean;
      onToggleShowAll: () => void;
      focusedSwatch: { scale: number; step: number };
      onFocusSwatch: (pos: { scale: number; step: number }) => void;
    }

    /**
     * OKLCH swatch matrix. Each swatch is inline-styled with its OKLCH values.
     * L/C/H axis labels visible; keyboard arrow-key navigation between swatches.
     */
    export function ColorSpecimen({
      scales,
      showAll,
      onToggleShowAll,
      focusedSwatch,
      onFocusSwatch,
    }: ColorSpecimenProps) {
      const visibleScales = showAll ? scales : scales.slice(0, 6);

      const handleKeyDown = (e: KeyboardEvent<HTMLDivElement>, scaleIdx: number, stepIdx: number) => {
        if (e.key === "ArrowRight") {
          e.preventDefault();
          onFocusSwatch({ scale: scaleIdx, step: Math.min(stepIdx + 1, visibleScales[scaleIdx].swatches.length - 1) });
        } else if (e.key === "ArrowLeft") {
          e.preventDefault();
          onFocusSwatch({ scale: scaleIdx, step: Math.max(stepIdx - 1, 0) });
        } else if (e.key === "ArrowDown") {
          e.preventDefault();
          onFocusSwatch({ scale: Math.min(scaleIdx + 1, visibleScales.length - 1), step: stepIdx });
        } else if (e.key === "ArrowUp") {
          e.preventDefault();
          onFocusSwatch({ scale: Math.max(scaleIdx - 1, 0), step: stepIdx });
        }
      };

      return (
        <div className="px-6 md:px-12 py-12 space-y-8">
          {/* Axis labels + show-all toggle */}
          <div className="flex items-center justify-between font-mono text-[var(--text-xs)] uppercase tracking-[0.15em] text-muted-foreground">
            <div className="flex gap-6">
              <span>L = LIGHTNESS</span>
              <span>C = CHROMA</span>
              <span>H = HUE</span>
            </div>
            <button
              onClick={onToggleShowAll}
              className="border border-foreground/30 px-3 py-1 uppercase tracking-[0.15em] hover:bg-foreground/10 transition-colors"
            >
              {showAll ? "CORE 6" : "SHOW ALL 49"}
            </button>
          </div>

          {/* Matrix */}
          <div className="space-y-2">

                <span className="font-mono text-[var(--text-xs)] uppercase text-foreground">{scale.name}</span>

                    return (
                      <div
                        key={sw.step}
                        data-oklch-swatch={`${scale.name}-${sw.step}`}

                        tabIndex={isFocused ? 0 : -1}
                        aria-label={`${scale.name} step ${sw.step} ${oklchStr}`}
                        onKeyDown={(e) => handleKeyDown(e, scaleIdx, stepIdx)}
                        onFocus={() => onFocusSwatch({ scale: scaleIdx, step: stepIdx })}

                        style={{ backgroundColor: oklchStr }}
                      />
                    );
                  })}
                </div>

            ))}
    ```

    **File D: `components/blocks/token-specimens/motion-specimen.tsx`** (Server Component — static SVG)

    ```tsx
    type MotionToken = { name: string; easing: string; duration: string; css: string };

    interface MotionSpecimenProps {
      tokens: MotionToken[];
    }

    /**
     * Each motion token rendered as an SVG cubic-bezier curve plot.
     * Shows the SHAPE of the motion — not a marketing animation.
     */
    export function MotionSpecimen({ tokens }: MotionSpecimenProps) {
      /** Parse "cubic-bezier(x1, y1, x2, y2)" → [x1, y1, x2, y2]. Fallback [0, 0, 1, 1] for "ease". */
      function parseCubicBezier(easing: string): [number, number, number, number] {
        const m = easing.match(/cubic-bezier\s*\(\s*([\d.-]+)\s*,\s*([\d.-]+)\s*,\s*([\d.-]+)\s*,\s*([\d.-]+)\s*\)/);
        if (m) return [parseFloat(m[1]), parseFloat(m[2]), parseFloat(m[3]), parseFloat(m[4])];
        // Named easings → approximate bezier control points
        const named: Record<string, [number, number, number, number]> = {
          "linear": [0, 0, 1, 1],
          "ease": [0.25, 0.1, 0.25, 1],
          "ease-in": [0.42, 0, 1, 1],
          "ease-out": [0, 0, 0.58, 1],
    ```
