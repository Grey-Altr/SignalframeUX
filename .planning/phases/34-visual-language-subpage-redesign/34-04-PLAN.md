    Build the surface groups once (useMemo) from `API_DOCS`:

    ```typescript
    type SurfaceGroup = { key: "COMPONENTS" | "HOOKS" | "TOKENS"; label: string; entries: Array<{ id: string; label: string }> };

    const SURFACE_GROUPS: SurfaceGroup[] = useMemo(() => {
      const groups: Record<SurfaceGroup["key"], Array<{ id: string; label: string }>> = {
        COMPONENTS: [],
        HOOKS: [],
        TOKENS: [],
      };
      for (const id in API_DOCS) {
        const doc = API_DOCS[id];
        if (doc.layer === "HOOK") groups.HOOKS.push({ id, label: doc.name });
        else if (doc.layer === "TOKEN") groups.TOKENS.push({ id, label: doc.name });
        else groups.COMPONENTS.push({ id, label: doc.name });  // CORE/FRAME/SIGNAL all → COMPONENTS
      }
      // Apply search filter
      const q = query.trim().toLowerCase();
      const filter = (arr: Array<{ id: string; label: string }>) =>
        q ? arr.filter((e) => e.id.toLowerCase().includes(q) || e.label.toLowerCase().includes(q)) : arr;
      return [
        { key: "COMPONENTS", label: "COMPONENTS", entries: filter(groups.COMPONENTS) },
        { key: "HOOKS", label: "HOOKS", entries: filter(groups.HOOKS) },
        { key: "TOKENS", label: "TOKENS", entries: filter(groups.TOKENS) },
      ];
    }, [query]);

    const visibleEntryCount = SURFACE_GROUPS.reduce((sum, g) => sum + g.entries.length, 0);
    ```
