**Create `eslint.config.js` in project root** with this exact structure:

```javascript
import { FlatCompat } from "@eslint/eslintrc";
import { dirname } from "path";
import { fileURLToPath } from "url";
import tsPlugin from "@typescript-eslint/eslint-plugin";
import tsParser from "@typescript-eslint/parser";

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);
const compat = new FlatCompat({ baseDirectory: __dirname });

/** @type {import("eslint").Linter.Config[]} */
const config = [
  // eslint-config-next via compat adapter (legacy format → flat)
  ...compat.extends("next/core-web-vitals"),

  // TypeScript files: @typescript-eslint strict type-checked
  {
    files: ["**/*.ts", "**/*.tsx"],
    plugins: { "@typescript-eslint": tsPlugin },
    languageOptions: {
      parser: tsParser,
      parserOptions: {
        project: true,
        tsconfigRootDir: __dirname,
      },
    },
    rules: {
      // recommended-type-checked rules (error level)
      ...tsPlugin.configs["flat/recommended-type-checked"].rules,
      // Style rules
      "@typescript-eslint/consistent-type-imports": ["error", { prefer: "type-imports" }],
    },
  },

  // Ignore patterns
  {
    ignores: [
      ".next/**",
      "node_modules/**",
      "public/**",
      ".planning/**",
    ],
  },
];

export default config;
```
