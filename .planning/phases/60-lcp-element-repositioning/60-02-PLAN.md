              width: rect.width,
              height: rect.height,
              sectionContain: cs ? cs.contain : null,
              sectionOverflow: cs ? cs.overflow : null,
            };
          });

          results.push({ viewport: vp.name, ...measurement! });
        });

        await fs.mkdir(path.dirname(OUTPUT_PATH), { recursive: true });
