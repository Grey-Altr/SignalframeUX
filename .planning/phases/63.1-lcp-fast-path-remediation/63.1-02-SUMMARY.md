## Pattern Applied (CRT-04 verbatim)

```typescript
// Synchronous guard BEFORE rIC (D-06):
if (window.matchMedia("(prefers-reduced-motion: reduce)").matches) return;

let ricHandle: number | undefined;
let cleanup: (() => void) | undefined;

const initAnimations = () => {
  ricHandle = undefined;
  // ... original GSAP/ScrollTrigger body ...
  cleanup = () => { trigger.kill(); /* etc */ };
};

type IdleCb = (cb: IdleRequestCallback, opts?: { timeout: number }) => number;
const ric = (window as Window & { requestIdleCallback?: IdleCb }).requestIdleCallback;
ricHandle = ric
  ? ric(initAnimations, { timeout: 100 })
  : (setTimeout(initAnimations, 0) as unknown as number);

return () => {
  const cancelRic = (window as Window & { cancelIdleCallback?: (h: number) => void }).cancelIdleCallback;
  if (ricHandle !== undefined) {
    if (cancelRic) cancelRic(ricHandle);
    else clearTimeout(ricHandle);
  }
  if (cleanup) cleanup();
};
```
