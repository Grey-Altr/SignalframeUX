test("zero React hydration warnings on showcase route", async ({ page }) => {
  const consoleMessages: { type: string; text: string }[] = [];
  page.on("console", (msg) => {
    consoleMessages.push({ type: msg.type(), text: msg.text() });
  });
  await page.goto(FIXTURE, { waitUntil: "networkidle" });
  // Wait an extra beat for any deferred hydration warnings to flush
  await page.waitForTimeout(250);
  const hydrationWarnings = consoleMessages.filter((m) =>
    /hydrat/i.test(m.text())
  );
  expect(
    hydrationWarnings,
    `Hydration warnings emitted: ${JSON.stringify(hydrationWarnings, null, 2)}`
  ).toEqual([]);
});

  // Click two day cells that are several days apart (avoid month edges to stay valid)

    .locator('[data-testid="fixture-uncontrolled-range"] [role="gridcell"] button:not([disabled]), [data-testid="fixture-uncontrolled-range"] td button:not([disabled])');

  await dayButtons.nth(7).click();
